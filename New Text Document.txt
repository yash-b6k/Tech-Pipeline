        // --- Core Navigation Logic ---
        function navigate(pageId, filterType = null) {

            // Show the requested page

        // Function to toggle a dropdown menu (used for desktop 'Job Notifications' and mobile)
        function toggleDropdown(dropdownId) {
            const dropdown = document.getElementById(dropdownId);
            if (dropdown) {
                // For all mobile dropdowns and the main desktop dropdown trigger
                if (dropdownId.startsWith('mobile') || dropdownId === 'jobs-dropdown') {
                    dropdown.classList.toggle('hidden');
                } else {
                    // This block is primarily for the desktop main dropdown's animation, 
                    // though for nested dropdowns, the CSS hover utilities handle visibility.
                    const isHidden = dropdown.classList.contains('hidden');
                    if (isHidden) {
                        dropdown.classList.remove('hidden', 'scale-95', 'opacity-0');
                        dropdown.classList.add('scale-100', 'opacity-100');
                    } else {
                        dropdown.classList.add('scale-95', 'opacity-0');
                        // Wait for transition before hiding completely
                        setTimeout(() => {
                            dropdown.classList.add('hidden');
                        }, 300);
                    }
                }
            }
        }

        // Function to render job cards based on category filter
        function renderJobCards(filterType) {
            const isHomePage = currentView === 'home';
            const containerId = isHomePage ? 'job-cards-container' : 'full-job-cards-container';
            const container = document.getElementById(containerId);
            const jobsTitle = document.getElementById('jobs-title');

            if (!container) return;

            // Determine the filter criteria
            let filteredJobs = jobData;
            let titleText = "All Job Opportunities";

            if (filterType) {
                if (filterType === 'IT' || filterType === 'Non-IT') {
                    filteredJobs = jobData.filter(job => job.type === filterType);
                    titleText = `${filterType} Jobs (Private Sector)`;
                } else if (filterType === 'Private') {
                    // Shows all private jobs (IT and Non-IT)
                    filteredJobs = jobData.filter(job => job.sector === 'Private');
                    titleText = "All Private Sector Jobs";
                } else if (filterType === 'Internships') {
                    filteredJobs = jobData.filter(job => job.sector === 'Internships');
                    titleText = "Internship Opportunities";
                } else if (filterType === 'Govt') {
                    filteredJobs = jobData.filter(job => job.sector === 'Govt');
                    titleText = "Government Jobs & Notifications";
                }
            } else if (isHomePage) {
                // For the home page, show a sample of 3 jobs
                filteredJobs = jobData.slice(0, 3);
            }

            // Update Jobs page title and description
            if (jobsTitle) {
                jobsTitle.textContent = titleText;
            }
            const jobsDescription = document.querySelector('#jobs-page p');
            if (jobsDescription) {
                jobsDescription.innerHTML = `Showing jobs filtered by **${filterType || 'All Categories'}**. Use the navigation menu to select a specific category.`;
            }

            // Generate HTML for all cards
            const cardHtml = filteredJobs.map(job => createJobCard(job)).join('');
            container.innerHTML = cardHtml;

            // Replace Lucide icons after rendering
            lucide.createIcons();
        }
